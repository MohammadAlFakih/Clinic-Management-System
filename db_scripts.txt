//Department Table
CREATE TABLE `clinic_db`.`department'
 (`id` INT NOT NULL AUTO_INCREMENT , `city` VARCHAR(25) NOT NULL , `details` TEXT NULL , `room` INT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;

//Doctor
CREATE TABLE `clinic_db`.`doctor` (`id` INT NOT NULL AUTO_INCREMENT , `department_id` INT NOT NULL ,
 `email` VARCHAR(30) NOT NULL , `password` TEXT NOT NULL , `first_name` VARCHAR(30) NOT NULL ,
 `last_name` VARCHAR(30) NOT NULL , `age` INT NOT NULL , `gender` CHAR(1) NOT NULL , `phone` VARCHAR(30) NOT NULL ,
 `specialization` VARCHAR(30) NOT NULL , `role` VARCHAR(30) NOT NULL DEFAULT 'doctor' , PRIMARY KEY (`id`), UNIQUE (`email`)) ENGINE = InnoDB;


//Doctor in department
ALTER TABLE `doctor` ADD CONSTRAINT `doctor_id_department` FOREIGN KEY (`department_id`)
 REFERENCES `department`(`id`) ON DELETE RESTRICT ON UPDATE CASCADE;

//Schedule
CREATE TABLE `clinic_db`.`schedule` (`id` INT NOT NULL AUTO_INCREMENT , `doctor_id` INT NOT NULL ,
 `start_date` DATE NOT NULL , `end_date` DATE NOT NULL , `sequence` VARCHAR(30) NOT NULL DEFAULT 'FFFFFFFFFFFFFFFFFFFF' ,
 PRIMARY KEY (`id`)) ENGINE = InnoDB;

//Doctor Schedule
ALTER TABLE `schedule` ADD CONSTRAINT `doctor_schedule` FOREIGN KEY (`doctor_id`)
 REFERENCES `doctor`(`id`) ON DELETE CASCADE ON UPDATE CASCADE;

//Secretary
CREATE TABLE `clinic_db`.`secretary` (`id` INT NOT NULL AUTO_INCREMENT , `doctor_id` INT NULL , `email` VARCHAR(30) NOT NULL ,
 `password` VARCHAR(255) NOT NULL , `first_name` VARCHAR(30) NOT NULL , `last_name` VARCHAR(30) NOT NULL , `age` INT NOT NULL ,
 `gender` CHAR(1) NOT NULL , `phone` VARCHAR(30) NOT NULL , `role` VARCHAR(30) NOT NULL DEFAULT 'secretary', PRIMARY KEY (`id`), UNIQUE (`email`)) ENGINE = InnoDB;

//Doctor Secretary
ALTER TABLE `secretary` ADD CONSTRAINT `doctor secretary` FOREIGN KEY (`doctor_id`) REFERENCES `doctor`(`id`) ON DELETE SET NULL ON UPDATE CASCADE;

//Patient
CREATE TABLE `clinic_db`.`patient` (`id` INT NOT NULL AUTO_INCREMENT , `email` VARCHAR(30) NOT NULL , `password` VARCHAR(255) NOT NULL ,
 `first_name` VARCHAR(30) NOT NULL , `last_name` VARCHAR(30) NOT NULL , `age` INT NOT NULL , `gender` CHAR(1) NOT NULL ,
 `phone` VARCHAR(30) NULL , `role` VARCHAR(30) NOT NULL DEFAULT 'patient' , PRIMARY KEY (`id`), UNIQUE (`email`)) ENGINE = InnoDB;

//Payment
CREATE TABLE `clinic_db`.`payment` (`patient_id` INT NOT NULL , `doctor_id` INT NOT NULL , `balance` INT NOT NULL ) ENGINE = InnoDB;

//Payment with doctor and patient
ALTER TABLE `payment` ADD CONSTRAINT `balance_with_doctor` FOREIGN KEY (`doctor_id`) REFERENCES `doctor`(`id`)
 ON DELETE RESTRICT ON UPDATE CASCADE; ALTER TABLE `payment` ADD CONSTRAINT `balance_with_patient` FOREIGN KEY (`patient_id`)
 REFERENCES `patient`(`id`) ON DELETE RESTRICT ON UPDATE CASCADE;

//Appointment
CREATE TABLE `clinic_db`. `appointment` (`id` INT NOT NULL AUTO_INCREMENT , `doctor_id` INT NOT NULL , `patient_id` INT NOT NULL ,
 `department_id` INT NOT NULL , `start_date` DATE NOT NULL , `end_date` DATE NOT NULL , `bill` INT NOT NULL ,
 `status` VARCHAR(30) NOT NULL DEFAULT 'upcoming' , PRIMARY KEY (`id`)) ENGINE = InnoDB;

//Appointment with doctor and patient in department
ALTER TABLE `appointment` ADD CONSTRAINT `doctor_appointment` FOREIGN KEY (`doctor_id`) REFERENCES `doctor`(`id`) 
ON DELETE RESTRICT ON UPDATE CASCADE;
ALTER TABLE `appointment` ADD CONSTRAINT `patient_appointment` FOREIGN KEY (`patient_id`) REFERENCES `patient`(`id`) 
ON DELETE RESTRICT ON UPDATE CASCADE; ALTER TABLE `appointment` ADD CONSTRAINT `appointment_place` FOREIGN KEY (`department_id`)
REFERENCES `department`(`id`) ON DELETE RESTRICT ON UPDATE NO ACTION;

//Document
CREATE TABLE `clinic_db`.`document` (`id` INT NOT NULL AUTO_INCREMENT , `appointment_id` INT NOT NULL ,
`details` TEXT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;

//Appointment Document
ALTER TABLE `document` ADD CONSTRAINT `appointment_document` FOREIGN KEY (`appointment_id`)
 REFERENCES `appointment`(`id`) ON DELETE CASCADE ON UPDATE CASCADE;

//Black List
CREATE TABLE `clinic_db`.`black_list` (`patient_id` INT NOT NULL ,
 `doctor_id` INT NOT NULL , `end_date` DATE NOT NULL ) ENGINE = InnoDB;

//Doctor block patient
ALTER TABLE `black_list` ADD CONSTRAINT `block_patient` FOREIGN KEY (`patient_id`) REFERENCES `patient`(`id`)
 ON DELETE CASCADE ON UPDATE CASCADE; ALTER TABLE `black_list` ADD CONSTRAINT `doctor_block` FOREIGN KEY (`doctor_id`)
 REFERENCES `doctor`(`id`) ON DELETE CASCADE ON UPDATE CASCADE;
